private void broadcastUpdate() {

private Ticket saveAndBroadcast(Ticket ticket) {

broadcastUpdate();
return saved;
}

@Transactional
public void processUserMessage(String phoneNumber, String messageType, String messageContent) {
// Check if sender is a staff member
Staff staff = staffRepository.findByPhoneNumber(phoneNumber).orElse(null);
if (staff != null) {
handleStaffWorkflow(staff, messageType, messageContent);
return;
}

// Regular client flow
Client client = clientRepository.findByPhoneNumber(phoneNumber)
.orElseGet(() -> createClient(phoneNumber));
ChatSession session = getOrCreateSession(phoneNumber);

// Test mode trigger (for developer testing)
if ("text".equals(messageType)) {
int testIndex = getTestModeIndex(messageContent);
if (testIndex > 0) {
enableTestMode(client, session, findActiveTicket(client), phoneNumber, testIndex);
return;
}
}

Ticket activeTicket = findActiveTicket(client);

if (activeTicket != null) {
handleActiveTicketFlow(client, session, activeTicket, messageType, messageContent, phoneNumber);
} else {
handleNewClientFlow(client, session, messageType, messageContent, phoneNumber);
}
}

// ============================================================
// NEW CLIENT FLOW — AIS Intelligence & Service Routing
// ============================================================
private void handleNewClientFlow(Client client, ChatSession session, String messageType,
String messageContent, String phoneNumber) {

if (!"text".equals(messageType)) {
if (session.getCurrentState() == null || session.getCurrentState() == ChatState.NEW) {
sendWelcomeNew(phoneNumber, session);
return;
}
}

ChatState state = session.getCurrentState() == null ? ChatState.NEW : session.getCurrentState();
String content = messageContent.trim();

switch (state) {
case NEW:
if (isGreeting(content)) {
session.setCurrentState(ChatState.COLLECTING_NAME);
sessionRepository.save(session);
send(phoneNumber, session, "👋 Welcome to Porwal CA\n\nI'm Arjun and I'll help you with your tax and compliance requirements today.\n\nTo get started, may I know your full name?");
} else {
session.setCurrentState(ChatState.COLLECTING_NAME);
sessionRepository.save(session);
send(phoneNumber, session, "👋 Welcome to Porwal CA\n\nI'm Arjun and I'll help you with your tax and compliance requirements today.\n\nTo get started, may I know your full name?");
}
break;

case COLLECTING_NAME:
if (content.length() > 2) {
session.setClientName(content);
session.setCurrentState(ChatState.COLLECTING_CITY);
sessionRepository.save(session);
send(phoneNumber, session, "Nice to meet you, " + content + ".\n\n📍 Which city are you currently based in?\n\nThis helps us assign the right CA expert for your case.");
} else {
send(phoneNumber, session, "Could you please share your full name to proceed?");
}
break;

case COLLECTING_CITY:
// Accepts text or location payload
session.setClientCity(content);
session.setCurrentState(ChatState.COLLECTING_PAN);
sessionRepository.save(session);
send(phoneNumber, session, "Perfect.\n\nBefore we proceed, could you please share your PAN number?\n\nWe'll use it only to identify your tax profile and relevant records.");
break;

case COLLECTING_PAN:
String pan = content.toUpperCase(Locale.ROOT).replaceAll("\\s+", "");
if (pan.matches("[A-Z]{5}[0-9]{4}[A-Z]{1}")) {
session.setCollectedPan(pan);
session.setItPortalId(pan); // Use PAN as ID
session.setCurrentState(ChatState.COLLECTING_DOB);
sessionRepository.save(session);
send(phoneNumber, session, "✅ PAN details recorded successfully.\n\nTo retrieve your AIS and tax records, please share your Date of Birth in DD-MM-YYYY format.");
} else {
send(phoneNumber, session, "Please provide a valid PAN format (e.g., ABCDE1234F).");
}
break;

case COLLECTING_DOB:
// Basic normalization to DDMMYYYY
String cleanDob = content.replaceAll("[^0-9]", "");
if (cleanDob.length() == 8) {
session.setClientDob(cleanDob);
session.setCurrentState(ChatState.COLLECTING_IT_PASSWORD);
sessionRepository.save(session);
send(phoneNumber, session, "Great.\n\nFor secure access to your tax information, we'll need your Income Tax Portal password.\n\nYour credentials are used only for the requested service and are handled confidentially.");
} else {
send(phoneNumber, session, "Please share your Date of Birth in the correct format (e.g., 01-01-1995).");
}
break;

case COLLECTING_IT_PASSWORD:
session.setItPortalPassword(content);
session.setCurrentState(ChatState.SERVICE_SELECTION_SHOWN);
sessionRepository.save(session);

send(phoneNumber, session, "✅ Verification details received.\n\nPlease tell us how we can assist you today:\n\n1️⃣ ITR Filing\n2️⃣ Tax Advisory\n3️⃣ GST Services\n4️⃣ Audit & Compliance\n5️⃣ Tax Notice / Appeal\n6️⃣ Financial Planning\n7️⃣ Business Registration\n\nReply with the option number or service name.");

// Trigger Silent Login and Full Scrape asynchronously
fetchAisDataAsync(phoneNumber, session);
break;

case AWAITING_OTP:
// Deprecated in new silent flow but kept for safety if needed later
session.setCurrentState(ChatState.SERVICE_SELECTION_SHOWN);
sessionRepository.save(session);
break;

case SERVICE_SELECTION_SHOWN:
String selectedService = "";
boolean isCallService = false;

if (content.contains("1") || content.toLowerCase().contains("itr")) {
selectedService = "ITR Filing";
} else if (content.contains("2") || content.toLowerCase().contains("audit")) {
selectedService = "Audit";
isCallService = true;
} else if (content.contains("3") || content.toLowerCase().contains("advisory") || content.toLowerCase().contains("appeal")) {
selectedService = "Tax Advisory";
isCallService = true;
} else if (content.contains("4") || content.toLowerCase().contains("financial")) {
selectedService = "Financial Planning";
isCallService = true;
} else if (content.contains("5") || content.toLowerCase().contains("consultancy")) {
selectedService = "Consultancy";
isCallService = true;
} else {
send(phoneNumber, session, "Kripya 1 se 5 ke beech koi option select karein.");
return;
}

// Set name if missing
if (isBlank(client.getName())) {
client.setName("Client " + phoneNumber.substring(phoneNumber.length() - 4));
clientRepository.save(client);
}

Ticket ticket = createTicket(client, selectedService, session);

if (isCallService) {
ticket.setStatus(TicketStatus.CALL_PENDING.name());
ticket.setTicketCategory("CALL_SERVICE");
saveAndBroadcast(ticket);

session.setCurrentState(ChatState.FINISHED);
sessionRepository.save(session);

send(phoneNumber, session, "✅ Your " + selectedService + " Consultation Has Been Scheduled\n\n" +
"One of our tax specialists will review your case and reach out shortly.\n\n" +
"If there's anything you'd like the CA to review beforehand, feel free to share it here.");
} else {
session.setExtractedService(selectedService);
session.setCurrentState(ChatState.AWAITING_DOCS);
sessionRepository.save(session);
sendDocumentChecklist(phoneNumber, session, ticket, true);
}
break;

default:
sendWelcomeNew(phoneNumber, session);
session.setCurrentState(ChatState.COLLECTING_PAN);
sessionRepository.save(session);
break;
}
}

private void fetchAisDataAsync(String phoneNumber, ChatSession session) {
log.info("Starting background Full Scrape for " + phoneNumber);

// ============================================================
// NOTIFY ADMIN: "Scraping in progress" via WebSocket broadcast
// ============================================================
broadcastUpdate(); // Refresh admin dashboard to show latest state

java.util.concurrent.CompletableFuture.runAsync(() -> {
try {
// We use triggerFullScrape which connects to our robust Python Scraper
Map<String, Object> scrapeResult = itPortalService.triggerFullScrape(
session.getCollectedPan(), session.getItPortalPassword(), session.getClientDob());

log.info("Scrape Result Status: " + scrapeResult.get("status"));

if (!"completed".equals(scrapeResult.get("status"))) {
whatsappMessageSender.sendMessage(phoneNumber, "❌ Income Tax Portal par login fail ho gaya ya data fetch nahi ho paya. Kripya apna sahi password wapas bhejein.");
session.setCurrentState(ChatState.COLLECTING_IT_PASSWORD);
sessionRepository.save(session);
return;
}

// Clear password for security
session.setItPortalPassword(null);

// ====================================================
// EXTRACT PROFILE DATA FROM SCRAPE AND UPDATE CLIENT
// ====================================================
@SuppressWarnings("unchecked")
Map<String, Object> profile = (Map<String, Object>) scrapeResult.get("profile");
Client c = clientRepository.findByPhoneNumber(phoneNumber).orElse(null);

if (profile != null && c != null) {
String scrapedName = (String) profile.get("name");
String scrapedPhone = (String) profile.get("phone");
String scrapedEmail = (String) profile.get("email");
String entityType = (String) profile.get("entity_type");

// Clean name: handle "ALCHEMIST MARKE... ALCHEMIST MARKETING TALENT SOLUTIONS PRIVATE LIMITED"
if (scrapedName != null && !scrapedName.isBlank()) {
scrapedName = cleanScrapedName(scrapedName);
if (scrapedName.length() > 250) {
scrapedName = scrapedName.substring(0, 250);
}
c.setName(scrapedName);
session.setClientName(scrapedName);
log.info("Updated client name from portal: " + scrapedName);
}

// Update Client Type from entity type
if (entityType != null && !entityType.isBlank()) {
c.setClientType(entityType);
}

clientRepository.save(c);
log.info("Client entity saved with portal data for: " + phoneNumber);
}

// Get the generated report
String reportPath = (String) scrapeResult.get("generated_report_path");
if (reportPath != null && !reportPath.isEmpty()) {
session.setCollectingField(null); // Don't store path in collectingField
log.info("Full Report PDF generated at: " + reportPath);

if (c != null) {
java.util.List<Ticket> tickets = ticketRepository.findAllByClientIdOrderByCreatedAtDesc(c.getId());
if (tickets != null && !tickets.isEmpty()) {
Ticket latestTicket = tickets.get(0);
latestTicket.setAisPdfPath(reportPath);

// Also store the AIS report JSON snapshot
try {
ObjectMapper mapper = new ObjectMapper();
String jsonSnapshot = mapper.writeValueAsString(scrapeResult);
latestTicket.setAisReportSnapshot(jsonSnapshot);
session.setAisReportJson(jsonSnapshot);
} catch (Exception jsonEx) {
log.warn("Could not serialize scrape result to JSON", jsonEx);
}

saveAndBroadcast(latestTicket);
}
}
}

sessionRepository.save(session);

// Notify user with scraped name
String clientNameDisplay = (c != null && c.getName() != null) ? c.getName() : "Client";
whatsappMessageSender.sendMessage(phoneNumber, 
"Got it. I've shared your tax profile information with our team for review.\n\n" +
"Your Pending Actions and Filed Forms are now available on the CA dashboard.");

// BROADCAST again so admin sees updated data with client name
broadcastUpdate();

} catch (Exception e) {
log.error("Failed to execute full scrape async", e);
whatsappMessageSender.sendMessage(phoneNumber, "We couldn't verify your credentials with the portal at this moment. Please double-check and provide your correct password.");
session.setCurrentState(ChatState.COLLECTING_IT_PASSWORD);
sessionRepository.save(session);
}
});
}

/**
* Cleans scraped name: handles truncation like "ALCHEMIST MARKE... ALCHEMIST MARKETING..."
* and removes Angular Material icon text.
*/
private String cleanScrapedName(String name) {
if (name == null || name.isBlank()) return name;

// Remove Angular Material icon ligatures
name = name.replaceAll("expand_more|expand_less|arrow_drop_down", "").trim();

// Handle "TRUNCATED... FULL NAME" pattern
if (name.contains("...")) {
String[] parts = name.split("\\.\\.\\.");
// Find the longest part — that's the full name
String longest = "";
for (String part : parts) {
String trimmed = part.trim();
if (trimmed.length() > longest.length()) {
longest = trimmed;
}
}
if (!longest.isBlank()) {
name = longest;
}
}

// Remove leading/trailing dots and spaces
name = name.replaceAll("^[.\\s]+|[.\\s]+$", "");

return name;
}

// ACTIVE TICKET FLOW — Client already has a ticket in progress
// ============================================================
private void handleActiveTicketFlow(Client client, ChatSession session, Ticket ticket,
String messageType, String messageContent, String phoneNumber) {
if ("location".equals(messageType)) {
handleLocationMessage(client, session, messageContent, phoneNumber);
return;
}

// Document/image upload
if ("document".equals(messageType) || "image".equals(messageType)) {
String status = ticket.getStatus();
if (TicketStatus.PENDING_ADMIN_APPROVAL.name().equals(status)
|| ChatState.AWAITING_DOCS.equals(session.getCurrentState())) {
verifyAndSaveClientDocument(session, ticket, messageContent, phoneNumber);
} else if (TicketStatus.AWAITING_PAYMENT.name().equals(status)) {
// Client sent payment screenshot
ticket.setPaymentProofUrl(messageContent);
ticket.setStatus(TicketStatus.PAYMENT_VERIFICATION_PENDING.name());
saveAndBroadcast(ticket);
send(phoneNumber, session, """
Screenshot receive ho gaya! 📸

Humari admin team payment verify karke aapka kaam shuru kar degi.
Aapko WhatsApp par update milega.
""");
} else {
send(phoneNumber, session, "📄 Document receive ho gaya. Team check karegi.");
}
return;
}

if (!"text".equals(messageType)) {
sendDocumentChecklist(phoneNumber, session, ticket, false);
return;
}

// ===========================================
// 🔐 CREDENTIAL DETECTION
// If admin requested credentials and client sends text containing id/password patterns
// ===========================================
if ("REQUESTED".equals(ticket.getCredentialStatus()) && "text".equals(messageType)) {
if (looksLikeCredentials(messageContent)) {
ticket.setCredentialData(messageContent.trim());
ticket.setCredentialStatus("RECEIVED");
ticket.setCredentialReceivedAt(java.time.LocalDateTime.now());
saveAndBroadcast(ticket);
String clientName = client.getName() != null ? client.getName() : "";
send(phoneNumber, session, String.format("""
%s, credentials receive ho gaye! ✅

Humari team aapki file process karega.
Kuch time mein update milega. 🙏
""", clientName));
log.info("Credentials received for ticket={} from phone={}", ticket.getId(), phoneNumber);
return;
}
}

// Update language preference silently
updateLanguagePreference(session, messageContent);

String status = ticket.getStatus();

if (TicketStatus.CALL_PENDING.name().equals(status)) {
send(phoneNumber, session, """
Aapki request already receive ho gayi hai. ✅

Hamare CA expert 24 ghante ke andar aapko call karenge.
Urgent note ho to yahin bhej dein, main team ko forward kar dunga.
""");
ticket.setClientRequestLog(appendLog(ticket.getClientRequestLog(), messageContent));
saveAndBroadcast(ticket);
return;
}

if (TicketStatus.PENDING_ADMIN_APPROVAL.name().equals(status)) {
// Still waiting for admin review
List<String> missing = missingDocuments(ticket, session);
if (!missing.isEmpty()) {
// Still needs docs
send(phoneNumber, session, String.format("""
%s, aapki file review queue mein hai.

Lekin kuch documents abhi bhi pending hain:
%s

Please clear photo ya PDF bhejein.
""", nullToDefault(client.getName(), ""), bulletList(missing)));
} else {
send(phoneNumber, session, String.format("""
%s, aapke saare documents submit ho gaye hain! ✅

Phorwal CA Firm team review kar rahi hai.
Fee aur payment link jald hi WhatsApp par milega.
""", nullToDefault(client.getName(), "")));
}
return;
}

if (TicketStatus.AWAITING_PAYMENT.name().equals(status)) {
String msg = messageContent.toLowerCase(Locale.ROOT).trim();
if (msg.length() > 5 && !msg.contains("hi") && !msg.contains("hello") && !msg.contains("help")) {
ticket.setPaymentProofUrl("TEXT: " + messageContent);
ticket.setStatus(TicketStatus.PAYMENT_VERIFICATION_PENDING.name());
saveAndBroadcast(ticket);
send(phoneNumber, session, String.format("""
%s, detail receive ho gayi! 🙏

Humari admin team payment verify karke aapka kaam shuru kar degi.
Aapko WhatsApp par update milega.
""", nullToDefault(client.getName(), "")));
} else {
send(phoneNumber, session, String.format("""
%s, payment abhi pending hai.

Kripya upar diye gaye QR code par payment karein aur yahan screenshot bhejein. 💳

Problem ho to reply karo — team help karegi.
""", nullToDefault(client.getName(), "")));
}
return;
}

if (TicketStatus.PAYMENT_VERIFICATION_PENDING.name().equals(status)) {
send(phoneNumber, session, "Aapki payment ki verification admin team kar rahi hai. Kripya thoda intezaar karein. 🙏");
return;
}

if (TicketStatus.PAYMENT_RECEIVED.name().equals(status)) {
send(phoneNumber, session, String.format("""
%s, payment confirm ho gaya! ✅

Team final verification ke baad kaam start karegi.
Jald hi update milega.
""", nullToDefault(client.getName(), "")));
return;
}

if (TicketStatus.IN_PROGRESS.name().equals(status)) {
send(phoneNumber, session, String.format("""
%s, aapka %s kaam chal raha hai! 🏃

Team actively work kar rahi hai.
Koi specific sawaal ho to yahin type karo.
""", nullToDefault(client.getName(), ""), ticket.getServiceType()));
return;
}

if (TicketStatus.ASSIGNED_TO_STAFF.name().equals(status)) {
// Forward to staff
forwardClientMessageToStaff(ticket, client, phoneNumber, messageContent);
return;
}

if (TicketStatus.PENDING_ADMIN_QC.name().equals(status)) {
send(phoneNumber, session, String.format("""
%s, aapka %s final quality check mein hai! 🔍

Thoda wait karo — final document ready hote hi yahin share kar diya jayega.
""", nullToDefault(client.getName(), ""), ticket.getServiceType()));
return;
}

if (TicketStatus.FINISHED.name().equals(status) || TicketStatus.COMPLETED.name().equals(status)) {
resetSession(session);
send(phoneNumber, session, String.format("""
%s, aapka previous %s complete ho chuka hai! 🎉

Nayi service ke liye batao — Phorwal CA Firm hamesha ready hai.
""", nullToDefault(client.getName(), ""), ticket.getServiceType()));
return;
}

// Default: send checklist
sendDocumentChecklist(phoneNumber, session, ticket, false);
}

// DOCUMENT VERIFICATION
// ============================================================
private void verifyAndSaveClientDocument(ChatSession session, Ticket ticket,
String mediaId, String phoneNumber) {
String cloudinaryUrl = whatsappMediaService.downloadAndSaveMedia(mediaId, phoneNumber);
if (cloudinaryUrl == null) {
send(phoneNumber, session, "❌ Document download/upload fail ho gaya. Please clear photo/PDF dobara bhejein.");
return;
}

boolean testMode = Boolean.TRUE.equals(session.getTestMode());
DocumentVerificationResult result = geminiService.verifyClientDocument(
cloudinaryUrl, ticket.getServiceType(), testMode);

if (!result.valid()) {
send(phoneNumber, session, String.format("""
❌ Is document ko verify nahi kar paya.

Reason: %s

Please sahi aur clear document dobara bhejein.

Required documents:
%s
""", nullToDefault(result.reason(), "Document unclear or wrong type."),
requiredDocsText(ticket, testMode)));
return;
}

appendClientDocument(ticket, result.documentType(), cloudinaryUrl);
appendVerifiedDocument(session, result.documentType());
saveAndBroadcast(ticket);
sessionRepository.save(session);

List<String> missingDocs = missingDocuments(ticket, session);
if (missingDocs.isEmpty()) {
// All docs submitted
send(phoneNumber, session, String.format("""
✅ %s verified and saved!

🎉 Saare required documents submit ho gaye hain!

Phorwal CA Firm team ab review karegi.
Fee aur payment link jald hi WhatsApp par milega.

Submitted documents:
%s
""", result.documentType(), documentLinksText(ticket)));
} else {
send(phoneNumber, session, String.format("""
✅ %s verified and saved!

Abhi ye documents pending hain:
%s

Please next document clear photo/PDF mein bhejein.
""", result.documentType(), bulletList(missingDocs)));
}
}

// STAFF WORKFLOW (via WhatsApp)
// ============================================================
private void handleStaffWorkflow(Staff staff, String messageType, String messageContent) {
List<Ticket> activeTickets = ticketRepository.findByAssignedStaffIdAndStatusIn(
staff.getId(), STAFF_ACTIVE_STATUSES);

if (activeTickets.isEmpty()) {
whatsappMessageSender.sendMessage(staff.getPhoneNumber(),
"ℹ️ Aapke paas abhi koi active assigned file nahi hai.\n\nAdmin assign karega tab notification aayega.");
return;
}

// Staff has exactly one ticket to work on (most common case)
Ticket workingTicket = activeTickets.get(0);
String clientPhoneNumber = workingTicket.getClient().getPhoneNumber();
String clientDisplay = nullToDefault(workingTicket.getClient().getName(), clientPhoneNumber);

// Document/image submission — submit for QC
if ("document".equals(messageType) || "image".equals(messageType)) {
String savedFilePath = whatsappMediaService.downloadAndSaveMedia(
messageContent, staff.getPhoneNumber());
if (savedFilePath == null) {
whatsappMessageSender.sendMessage(staff.getPhoneNumber(),
"❌ Document upload fail ho gaya. Dobara try karo.");
return;
}
workingTicket.setStaffSubmittedDocument(savedFilePath);
workingTicket.setStatus(TicketStatus.PENDING_ADMIN_QC.name());
workingTicket.setProgressPercent(90);
saveAndBroadcast(workingTicket);
whatsappMessageSender.sendMessage(staff.getPhoneNumber(),
"✅ Work submitted for admin QC review.\n\nAdmin review ke baad client ko deliver kar dega.");
return;
}

if (!"text".equals(messageType)) {
whatsappMessageSender.sendMessage(staff.getPhoneNumber(),
"Commands:\nSTATUS — current file info\nREQUEST: [document name] — client se maango\nQUERY: [question] — client ko puchho\nDONE — work complete (final PDF upload karo)\n\nOr final PDF/image directly upload karo.");
return;
}

String input = messageContent.trim();
String inputUpper = input.toUpperCase(Locale.ROOT);

if (inputUpper.equals("STATUS")) {
whatsappMessageSender.sendMessage(staff.getPhoneNumber(), staffStatusMessage(workingTicket));

} else if (inputUpper.startsWith("REQUEST:")) {
String needed = input.substring(input.indexOf(":") + 1).trim();
String clientMsg = String.format("📋 CA team ko aapki file ke liye ye chahiye:\n\n%s\n\nPlease yahin WhatsApp par bhejein.", needed);
whatsappMessageSender.sendMessage(clientPhoneNumber, clientMsg);
whatsappMessageSender.sendMessage(staff.getPhoneNumber(), "✅ Request sent to client: " + clientDisplay);

} else if (inputUpper.startsWith("QUERY:")) {
String query = input.substring(input.indexOf(":") + 1).trim();
String clientMsg = String.format("❓ CA team ka sawaal:\n\n%s\n\nPlease yahin reply karein.", query);
whatsappMessageSender.sendMessage(clientPhoneNumber, clientMsg);
whatsappMessageSender.sendMessage(staff.getPhoneNumber(), "✅ Query sent to " + clientDisplay);

} else if (inputUpper.startsWith("UPDATE:")) {
// Progress update from staff
String update = input.substring(input.indexOf(":") + 1).trim();
workingTicket.setStaffUpdate(appendLog(workingTicket.getStaffUpdate(), update));
saveAndBroadcast(workingTicket);
whatsappMessageSender.sendMessage(staff.getPhoneNumber(), "✅ Update saved.");

} else if (inputUpper.equals("DONE")) {
whatsappMessageSender.sendMessage(staff.getPhoneNumber(),
"✅ Work done record ho gaya.\n\nFinal PDF/image upload karo taaki admin QC ke baad client ko deliver ho sake.");

} else {
// Try to forward to client as a general message
whatsappMessageSender.sendMessage(staff.getPhoneNumber(),
"Commands:\nSTATUS — file info dekhne ke liye\nREQUEST: [info] — client se document maangne ke liye\nQUERY: [question] — client ko sawaal poochne ke liye\nUPDATE: [progress] — progress note karne ke liye\n\nFinal work ke liye document/PDF upload karo.");
}
}
