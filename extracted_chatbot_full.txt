}

// ============================================================
// NEW CLIENT FLOW — AIS Intelligence & Service Routing
// ============================================================
private void handleNewClientFlow(Client client, ChatSession session, String messageType,
String messageContent, String phoneNumber) {

if (!"text".equals(messageType)) {
if (session.getCurrentState() == null || session.getCurrentState() == ChatState.NEW) {
sendWelcomeNew(phoneNumber, session);
return;
}
}

ChatState state = session.getCurrentState() == null ? ChatState.NEW : session.getCurrentState();
String content = messageContent.trim();

switch (state) {
case NEW:
if (client.getName() != null && !client.getName().isBlank()) {
// Returning User Flow
session.setClientName(client.getName());
session.setCurrentState(ChatState.SERVICE_SELECTION_SHOWN);
sessionRepository.save(session);
send(phoneNumber, session, "Welcome back " + client.getName() + " 👋\n\nHow can we assist you today?\n\n1️⃣ ITR Filing\n2️⃣ GST Services\n3️⃣ Notice / Appeal\n4️⃣ Tax Advisory");
} else {
// New User Flow
session.setCurrentState(ChatState.COLLECTING_NAME);
sessionRepository.save(session);
send(phoneNumber, session, "✨ Welcome to Porwal CA\n\nI'm Arjun and I'll help you with your tax and compliance requirements today.\n\nTo get started, may I know your full name?");
}
break;

case COLLECTING_NAME:
if (content.length() > 2) {
session.setClientName(content);
session.setCurrentState(ChatState.COLLECTING_CITY);
sessionRepository.save(session);
send(phoneNumber, session, "Nice to meet you, " + content + ".\n\n📍 Which city are you currently based in?\n\nThis helps us assign the right CA expert for your case.");
} else {
send(phoneNumber, session, "Could you please share your full name to proceed?");
}
break;

case COLLECTING_CITY:
// Accepts text or location payload
session.setClientCity(content);
session.setCurrentState(ChatState.COLLECTING_PAN);
sessionRepository.save(session);
send(phoneNumber, session, "Got it.\n\nNow, could you please share your **PAN Number**?");
break;

case COLLECTING_PAN:
session.setClientPan(content.toUpperCase());
session.setCurrentState(ChatState.COLLECTING_DOB);
sessionRepository.save(session);
send(phoneNumber, session, "Thank you. Lastly, what is your **Date of Birth**? (Format: DD/MM/YYYY)");
break;

case COLLECTING_DOB:
session.setClientDob(content);
session.setCurrentState(ChatState.COLLECTING_PASSWORD);
sessionRepository.save(session);
send(phoneNumber, session, "Great. Now please provide your **IT Portal Password** (it will be securely stored for our CA team to process your return).");
break;

case COLLECTING_PASSWORD:
session.setClientItPassword(content);
session.setCurrentState(ChatState.SERVICE_SELECTION_SHOWN);
sessionRepository.save(session);
send(phoneNumber, session, "Perfect.\n\nPlease tell us how we can assist you today:\n\n1️⃣ ITR Filing\n2️⃣ GST Services\n3️⃣ Notice / Appeal\n4️⃣ Tax Advisory");
break;

case SERVICE_SELECTION_SHOWN:
String selectedService = "";
boolean isCallService = false;

if (content.contains("1") || content.toLowerCase().contains("itr")) {
selectedService = "ITR Filing";
isCallService = false;
} else if (content.contains("2") || content.toLowerCase().contains("gst")) {
selectedService = "GST Services";
isCallService = false;
} else if (content.contains("3") || content.toLowerCase().contains("notice") || content.toLowerCase().contains("appeal")) {
selectedService = "Tax Notice / Appeal";
isCallService = false;
} else if (content.contains("4") || content.toLowerCase().contains("advisory")) {
selectedService = "Tax Advisory";
isCallService = true;
} else {
send(phoneNumber, session, "Kripya 1 se 4 ke beech koi option select karein.");
return;
}

// Set name if missing
if (isBlank(client.getName())) {
client.setName(session.getClientName() != null ? session.getClientName() : "Client " + phoneNumber.substring(phoneNumber.length() - 4));
}
if (isBlank(client.getCity()) && session.getClientCity() != null) {
client.setCity(session.getClientCity());
}
if (isBlank(client.getPan()) && session.getClientPan() != null) {
client.setPan(session.getClientPan());
}
if (isBlank(client.getDob()) && session.getClientDob() != null) {
client.setDob(session.getClientDob());
}
if (isBlank(client.getItPassword()) && session.getClientItPassword() != null) {
client.setItPassword(session.getClientItPassword());
}
clientRepository.save(client);

session.setExtractedService(aiResponse.getService());
}

boolean hasCity = client.getCity() != null && !client.getCity().isEmpty();
boolean hasService = session.getExtractedService() != null && !session.getExtractedService().isEmpty();

if (hasCity && hasService && session.getCurrentState() != ChatState.AWAITING_DOCS) {
Ticket newTicket = new Ticket();
newTicket.setClient(client);
newTicket.setServiceType(session.getExtractedService());
newTicket.setStatus("PENDING_ADMIN_APPROVAL");
ticketRepository.save(newTicket);

session.setCurrentState(ChatState.AWAITING_DOCS);
sessionRepository.save(session);

System.out.println("🎟️ Ticket created -> Service: " + session.getExtractedService() + " | City: " + client.getCity());
whatsappMessageSender.sendMessage(phoneNumber, aiResponse.getReply_message());

callTicket.setClientRequestLog(content);
saveAndBroadcast(callTicket);

session.setCurrentState(ChatState.FINISHED);
sessionRepository.save(session);

send(phoneNumber, session, "✅ Your " + session.getExtractedService() + " Consultation Has Been Scheduled\n\n" +
"One of our tax specialists will review your case and reach out shortly.");
break;

default:
sendWelcomeNew(phoneNumber, session);
session.setCurrentState(ChatState.NEW);
sessionRepository.save(session);
break;
}
}

// ACTIVE TICKET FLOW — Client already has a ticket in progress
// ============================================================
private void handleActiveTicketFlow(Client client, ChatSession session, Ticket ticket,
String messageType, String messageContent, String phoneNumber) {
if ("location".equals(messageType)) {
handleLocationMessage(client, session, messageContent, phoneNumber);
return;
}

// Document/image upload
if ("document".equals(messageType) || "image".equals(messageType)) {
String status = ticket.getStatus();
if (TicketStatus.PENDING_ADMIN_APPROVAL.name().equals(status)
|| ChatState.AWAITING_DOCS.equals(session.getCurrentState())
|| TicketStatus.NORMAL_FLOW.name().equals(status)) {
verifyAndSaveClientDocument(session, ticket, messageContent, phoneNumber);
} else if (TicketStatus.WAITING_FOR_CLIENT_DOCUMENT.name().equals(status)) {
handleCustomDocumentUpload(session, ticket, messageContent, phoneNumber);
} else if (TicketStatus.AWAITING_PAYMENT.name().equals(status)) {
// Client sent payment screenshot
ticket.setPaymentProofUrl(messageContent);
ticket.setStatus(TicketStatus.PAYMENT_VERIFICATION_PENDING.name());
saveAndBroadcast(ticket);
send(phoneNumber, session, """
Screenshot receive ho gaya! ✅

Humari admin team payment verify karke aapka kaam shuru kar degi.
Aapko WhatsApp par update milega.
""");
} else {
send(phoneNumber, session, "✅ Document receive ho gaya. Team check karegi.");
}
return;
}

if (!"text".equals(messageType)) {
sendDocumentChecklist(phoneNumber, session, ticket, false);
return;
}

// ===========================================
// 🔐 CREDENTIAL DETECTION
// If admin requested credentials and client sends text containing id/password patterns
// ===========================================
if ("REQUESTED".equals(ticket.getCredentialStatus()) && "text".equals(messageType)) {
if (looksLikeCredentials(messageContent)) {
ticket.setCredentialData(messageContent.trim());
ticket.setCredentialStatus("RECEIVED");
ticket.setCredentialReceivedAt(java.time.LocalDateTime.now());
saveAndBroadcast(ticket);
String clientName = client.getName() != null ? client.getName() : "";
send(phoneNumber, session, String.format("""
%s, credentials receive ho gaye! ✅

Humari team aapki file process karega.
Kuch time mein update milega. 🙏
""", clientName));
log.info("Credentials received for ticket={} from phone={}", ticket.getId(), phoneNumber);
return;
}
}

// Update language preference silently
updateLanguagePreference(session, messageContent);

String status = ticket.getStatus();

if (TicketStatus.CALL_PENDING.name().equals(status)) {
send(phoneNumber, session, """
Aapki request already receive ho gayi hai. ✅

Hamare CA expert 24 ghante ke andar aapko call karenge.
Urgent note ho to yahin bhej dein, main team ko forward kar dunga.
""");
ticket.setClientRequestLog(appendLog(ticket.getClientRequestLog(), messageContent));
saveAndBroadcast(ticket);
return;
}

if (TicketStatus.PENDING_ADMIN_APPROVAL.name().equals(status)) {
// Still waiting for admin review
List<String> missing = missingDocuments(ticket, session);
if (!missing.isEmpty()) {
// Still needs docs
send(phoneNumber, session, String.format("""
%s, aapki file review queue mein hai.

Lekin kuch documents abhi bhi pending hain:
%s

Please clear photo ya PDF bhejein.
""", nullToDefault(client.getName(), ""), bulletList(missing)));
} else {
send(phoneNumber, session, String.format("""
%s, aapke saare documents submit ho gaye hain! ✅

Phorwal CA Firm team review kar rahi hai.
Fee aur payment link jald hi WhatsApp par milega.
""", nullToDefault(client.getName(), "")));
}
return;
}

if (TicketStatus.AWAITING_PAYMENT.name().equals(status)) {
String msg = messageContent.toLowerCase(Locale.ROOT).trim();
if (msg.length() > 5 && !msg.contains("hi") && !msg.contains("hello") && !msg.contains("help")) {
ticket.setPaymentProofUrl("TEXT: " + messageContent);
ticket.setStatus(TicketStatus.PAYMENT_VERIFICATION_PENDING.name());
saveAndBroadcast(ticket);
send(phoneNumber, session, String.format("""
%s, detail receive ho gayi! 🙏

Humari admin team payment verify karke aapka kaam shuru kar degi.
Aapko WhatsApp par update milega.
""", nullToDefault(client.getName(), "")));
} else {
send(phoneNumber, session, String.format("""
%s, payment abhi pending hai.

Kripya upar diye gaye QR code par payment karein aur yahan screenshot bhejein. 💳

Problem ho to reply karo — team help karegi.
""", nullToDefault(client.getName(), "")));
}
return;
}

if (TicketStatus.PAYMENT_VERIFICATION_PENDING.name().equals(status)) {
send(phoneNumber, session, "Aapki payment ki verification admin team kar rahi hai. Kripya thoda intezaar karein. 🙏");
return;
}

if (TicketStatus.PAYMENT_RECEIVED.name().equals(status)) {
send(phoneNumber, session, String.format("""
%s, payment confirm ho gaya! ✅

Team final verification ke baad kaam start karegi.
Jald hi update milega.
""", nullToDefault(client.getName(), "")));
return;
}

if (TicketStatus.IN_PROGRESS.name().equals(status)) {
send(phoneNumber, session, String.format("""
%s, aapka %s kaam chal raha hai! 🏃

Team actively work kar rahi hai.
Koi specific sawaal ho to yahin type karo.
""", nullToDefault(client.getName(), ""), ticket.getServiceType()));
return;
}

if (TicketStatus.ASSIGNED_TO_STAFF.name().equals(status)) {
// Forward to staff
forwardClientMessageToStaff(ticket, client, phoneNumber, messageContent);
return;
}

if (TicketStatus.PENDING_ADMIN_QC.name().equals(status)) {
send(phoneNumber, session, String.format("""
%s, aapka %s final quality check mein hai! 🔍

Thoda wait karo — final document ready hote hi yahin share kar diya jayega.
""", nullToDefault(client.getName(), ""), ticket.getServiceType()));
return;
}

if (TicketStatus.FINISHED.name().equals(status) || TicketStatus.COMPLETED.name().equals(status)) {
resetSession(session);
send(phoneNumber, session, String.format("""
%s, aapka previous %s complete ho chuka hai! 🎉

Nayi service ke liye batao — Phorwal CA Firm hamesha ready hai.
""", nullToDefault(client.getName(), ""), ticket.getServiceType()));
return;
}

// Default: send checklist
sendDocumentChecklist(phoneNumber, session, ticket, false);
}

// DOCUMENT VERIFICATION
// ============================================================
private void handleCustomDocumentUpload(ChatSession session, Ticket ticket, String mediaId, String phoneNumber) {
String cloudinaryUrl = whatsappMediaService.downloadAndSaveMedia(mediaId, phoneNumber);
if (cloudinaryUrl == null) {
send(phoneNumber, session, "⚠ Document download failed. Please send again.");
return;
}

java.util.List<com.caCommand.caCommand.entities.CustomDocumentRequest> requests = customDocumentRequestRepository.findByTicketIdAndStatus(ticket.getId(), "PENDING");
if (requests.isEmpty()) {
send(phoneNumber, session, "✅ Document received. Our team will review it.");
return;
}

// Just take the first pending request for now
com.caCommand.caCommand.entities.CustomDocumentRequest req = requests.get(0);
DocumentVerificationResult result = geminiService.verifyCustomDocument(cloudinaryUrl, req.getDocumentName());

if (!result.valid()) {
send(phoneNumber, session, String.format("⚠ Uploaded document does not match requested item (%s).\n\nReason: %s", req.getDocumentName(), result.reason()));
return;
}

req.setStatus("FULFILLED");
req.setFulfilledAt(java.time.LocalDateTime.now());
customDocumentRequestRepository.save(req);

// Append to ticket documents
appendClientDocument(ticket, req.getDocumentName(), cloudinaryUrl);
ticket.setStatus(TicketStatus.UNDER_REVIEW.name()); // move back to review
saveAndBroadcast(ticket);

send(phoneNumber, session, String.format("✅ %s received and verified.\n\nYour case has been updated successfully.", req.getDocumentName()));
}

private void verifyAndSaveClientDocument(ChatSession session, Ticket ticket,
String mediaId, String phoneNumber) {
String cloudinaryUrl = whatsappMediaService.downloadAndSaveMedia(mediaId, phoneNumber);
if (cloudinaryUrl == null) {
send(phoneNumber, session, "? Document download/upload fail ho gaya. Please clear photo/PDF dobara bhejein.");
return;
}

// 1. Verify instantly using AI
com.caCommand.caCommand.dtos.DocumentVerificationResult result = geminiService.verifyClientDocument(cloudinaryUrl, ticket.getServiceType(), false);

if (!result.valid()) {
send(phoneNumber, session, "? " + result.reason() + "\n\nKripya sahi document bhejein.");
return;
}

String verifiedDoc = geminiService.canonicalizeDocumentName(result.documentType());

synchronized (ticket.getId().toString().intern()) {
Ticket freshTicket = ticketRepository.findById(ticket.getId()).orElse(ticket);

// Handle client documents tracking with "Name :: URL" format
java.util.Map<String, String> collectedDocs = new java.util.LinkedHashMap<>();
if (freshTicket.getClientDocuments() != null && !freshTicket.getClientDocuments().isEmpty()) {
String[] lines = freshTicket.getClientDocuments().split("\n");
for (String line : lines) {
String[] parts = line.split(" :: ");
if (parts.length >= 1) {
// Filter out old JSON brackets if they exist from before
String name = parts[0].trim().replace("[\"", "").replace("\"]", "");
String url = parts.length >= 2 ? parts[1].trim() : "";
collectedDocs.put(name, url);
}
}
}

collectedDocs.put(verifiedDoc, cloudinaryUrl);

StringBuilder sbDocs = new StringBuilder();
List<String> collected = new ArrayList<>();
for (java.util.Map.Entry<String, String> entry : collectedDocs.entrySet()) {
sbDocs.append(entry.getKey()).append(" :: ").append(entry.getValue()).append("\n");
collected.add(entry.getKey());
}
freshTicket.setClientDocuments(sbDocs.toString());

saveAndBroadcast(freshTicket);
}

// Generate checklist status message
Runnable sendTask = () -> {
try {
Ticket latestTicket = ticketRepository.findById(ticket.getId()).orElse(ticket);
List<String> reqDocs = geminiService.getRequiredDocuments(latestTicket.getServiceType());
List<String> coll = new ArrayList<>();
if (latestTicket.getClientDocuments() != null && !latestTicket.getClientDocuments().isEmpty()) {
for (String line : latestTicket.getClientDocuments().split("\n")) {
String[] parts = line.split(" :: ");
if (parts.length >= 1) {
coll.add(parts[0].trim().replace("[\"", "").replace("\"]", ""));
}
}
}

List<String> miss = new ArrayList<>(reqDocs);
miss.removeAll(coll);

StringBuilder sbuild = new StringBuilder();
sbuild.append("\u2705 Document received and verified.\n\n");
sbuild.append("\uD83D\uDCCA *Document Status*\n\n");

sbuild.append("*Received:*\n");
for (String c : coll) {
sbuild.append("\u2713 ").append(c).append("\n");
}

if (!miss.isEmpty()) {
sbuild.append("\n*Missing:*\n");
for (String m : miss) {
sbuild.append("\u26A0\uFE0F ").append(m).append("\n");
}
}
send(phoneNumber, session, sbuild.toString());
} catch (Exception e) {
log.error("Error sending debounced message", e);
}
};

java.util.concurrent.ScheduledFuture<?> existing = pendingMessages.get(phoneNumber);
if (existing != null) {
existing.cancel(false);
}
pendingMessages.put(phoneNumber, scheduler.schedule(sendTask, 4, java.util.concurrent.TimeUnit.SECONDS));

// 2. Start Async Pipeline Process using Event-Driven Architecture (Phase 1)

private void handleLocationMessage(Client client, ChatSession session,
String messageContent, String phoneNumber) {
String[] coords = messageContent.split(",");
if (coords.length != 2) {
send(phoneNumber, session, "📍 Location read nahi ho payi. Please apni city name type karo, jaise \"Ajmer\".");
return;
}

String detectedCity = locationService.getCityFromCoordinates(coords[0].trim(), coords[1].trim());
if (isBlank(detectedCity)) {
send(phoneNumber, session, "📍 City detect nahi ho payi. Please apni city type karo.");
return;
}

client.setCity(detectedCity);
clientRepository.save(client);

if (!isBlank(session.getExtractedService())) {
Ticket ticket = findActiveTicket(client);
if (ticket == null) {
ticket = createTicket(client, session.getExtractedService(), session);
session.setCurrentState(ChatState.AWAITING_DOCS);
session.setVerifiedDocumentTypes("");
sessionRepository.save(session);
}
sendDocumentChecklist(phoneNumber, session, ticket, true);
} else {
String name = nullToDefault(client.getName(), "");
send(phoneNumber, session, String.format("""
📍 %s, aap %s se hain — noted!

Ab batao kaunsi CA service chahiye:
ITR Filing, GST, Balance Sheet, Company Registration, etc.
""", isBlank(name) ? "" : name + " ", detectedCity));
}
}

private boolean isBlank(String value) {
return value == null || value.trim().isEmpty();
}

private String nullToDefault(String value, String defaultValue) {
return isBlank(value) ? defaultValue : value;
}

private List<String> csvToList(String value) {
if (isBlank(value)) return new ArrayList<>();
String[] parts = value.split(",");
List<String> result = new ArrayList<>();
for (String part : parts) {
if (!isBlank(part)) result.add(part.trim());
}
return result;
}

private String normalize(String value) {
return value == null ? "" : value.toLowerCase(Locale.ROOT).replaceAll("[^a-z0-9]", "");
}

private String appendLog(String current, String entry) {
if (isBlank(entry)) return current;
String stamped = java.time.LocalDateTime.now() + " - " + entry.trim();
return isBlank(current) ? stamped : current + "\n" + stamped;
}

/**
* Detects if a WhatsApp message looks like login credentials.
* Matches patterns like:
*   "ID: abc@email.com Password: 12345"
*   "login: xxx pass: yyy"
*   "username abc password xyz"
*   "user: abc@email pwd: pass123"
*/
private boolean looksLikeCredentials(String message) {
if (message == null || message.isBlank()) return false;
String lower = message.toLowerCase(Locale.ROOT);
// Must contain both an identifier keyword AND a password keyword
boolean hasIdKeyword = lower.contains("id:") || lower.contains("user:") || lower.contains("login:")
|| lower.contains("username:") || lower.contains("userid:") || lower.contains("user id")
|| lower.contains("email:") || lower.contains("mobile:") || lower.contains("pan:")
|| (lower.contains("id") && lower.contains("@")); // email pattern
boolean hasPassKeyword = lower.contains("password:") || lower.contains("pass:") || lower.contains("pwd:")
|| lower.contains("passcode:") || lower.contains("password ")
|| lower.contains("pass ") || lower.contains("otp:");
return hasIdKeyword && hasPassKeyword;
}
